Optimización de Backend

Esta sección reúne las herramientas más potentes del motor Iridium para construir APIs completas: el sistema de rutas web, las clases y objetos, y las funciones de respuesta jsonify() y html().

1. Sistema de Rutas (`route` + `serve`)

Define endpoints HTTP de forma declarativa. Cada bloque route recibe automáticamente tres objetos inyectados: body (datos POST/PUT), params (query string) y path_params (segmentos de la URL).

// Ruta GET simple
route "/estado" => "GET":
    return jsonify({status: "ok", version: "2.0"})
end

// Ruta con parámetro dinámico en la URL
route "/usuario/<id>" => "GET":
    var user = db.query_one("SELECT * FROM usuarios WHERE id = ?", id)
    return jsonify({nombre: user.nombre, email: user.email})
end

// Ruta POST que acepta múltiples métodos
route "/login" => "POST":
    var user = db.query_one("SELECT * FROM usuarios WHERE nombre = ?", body.nombre)
    var ok = check_password(user.password, body.password)
    return jsonify({autenticado: ok})
end

serve(8080)

2. Clases y Objetos

Iridium soporta programación orientada a objetos mediante la palabra clave class. Los campos se declaran con var y los métodos con function. Dentro de un método, los campos de la instancia son accesibles directamente por su nombre.

class Usuario():
    var nombre = ""
    var email = ""
    var exp = 0

    function saludar():
        out "Hola, soy {nombre} y tengo {exp} puntos"
    end

    function ganar_exp(cantidad):
        exp += cantidad
    end
end

var u = Usuario()
u.nombre = "Zadquiel"
u.ganar_exp(50)
u.saludar() // → "Hola, soy Zadquiel y tengo 50 puntos"

3. Respuestas Web: `jsonify()` y `html()`

Dentro de un bloque route, tienes dos funciones nativas para generar respuestas HTTP. jsonify() serializa cualquier objeto a JSON, y html() devuelve contenido HTML con el tipo de contenido correcto.

// JSON con código de estado personalizado
return jsonify({error: "No autorizado"}, 401)

// Respuesta HTML
return html("<h1>Bienvenido a Iridium</h1>")

4. Seguridad: Hash de Contraseñas

El motor expone dos funciones globales para el manejo seguro de contraseñas, disponibles en cualquier punto del script:

// Al registrar un usuario
var hash = hash_password("mi_clave_secreta")
db.exec("INSERT INTO usuarios (password) VALUES (?)", hash)

// Al verificar el login
var user = db.query_one("SELECT password FROM usuarios WHERE nombre = ?", body.nombre)
var es_valido = check_password(user.password, body.clave)
return jsonify({ok: es_valido})
